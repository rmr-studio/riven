    ```kotlin
    @Service
    class CustomSourceConnectionService(
        private val logger: KLogger,
        private val repository: CustomSourceConnectionRepository,
        private val encryptionService: CredentialEncryptionService,
        private val ssrfValidator: SsrfValidatorService,
        private val roVerifier: ReadOnlyRoleVerifierService,
        private val authTokenService: AuthTokenService,
        private val activityService: ActivityService,
        private val objectMapper: ObjectMapper,
    ) {
        // ------ Public mutations ------

        @Transactional
        @PreAuthorize("@workspaceSecurity.hasWorkspace(#request.workspaceId)")
        fun create(request: CreateCustomSourceConnectionRequest): CustomSourceConnectionModel {
            val userId = authTokenService.getUserId()
            runGateChain(request.host, request.port, request.database, request.user, request.password, request.sslMode.value)
            val encrypted = encryptPayload(toPayload(request))
            val saved = repository.save(CustomSourceConnectionEntity(
                workspaceId = request.workspaceId,
                name = request.name,
                connectionStatus = ConnectionStatus.CONNECTED,
                encryptedCredentials = encrypted.ciphertext,
                iv = encrypted.iv,
                keyVersion = encrypted.keyVersion,
                lastVerifiedAt = ZonedDateTime.now(),
            ))
            activityService.logActivity(... /* create */)
            return saved.toModel(request.host, request.port, request.database, request.user, request.sslMode.value)
        }

        @Transactional(readOnly = true)
        fun test(request: TestCustomSourceConnectionRequest): TestResult {
            // Aggregate status with category — chose aggregate per research open-question, per-gate as follow-up
            return try {
                runGateChain(request.host, request.port, request.database, request.user, request.password, request.sslMode.value)
                TestResult(pass = true, category = null, message = "All gates passed")
            } catch (e: SsrfRejectedException) { TestResult(false, "SSRF", e.message ?: "") }
              catch (e: ReadOnlyVerificationException) { TestResult(false, "RO_VERIFY", sanitizeForUser(e.message)) }
        }

        @Transactional
        @PreAuthorize("@workspaceSecurity.hasWorkspace(#workspaceId)")
        fun update(workspaceId: UUID, id: UUID, request: UpdateCustomSourceConnectionRequest): CustomSourceConnectionModel { ... }

        @Transactional
        @PreAuthorize("@workspaceSecurity.hasWorkspace(#workspaceId)")
        fun softDelete(workspaceId: UUID, id: UUID) {
            val entity = findOrThrow { repository.findByIdAndWorkspaceId(id, workspaceId) }
            entity.deleted = true
            entity.deletedAt = ZonedDateTime.now()
            repository.save(entity)
            activityService.logActivity(... /* delete */)
        }

        // ------ Public reads ------

        @Transactional(readOnly = true)
        @PreAuthorize("@workspaceSecurity.hasWorkspace(#workspaceId)")
        fun getById(workspaceId: UUID, id: UUID): CustomSourceConnectionModel {
            val entity = findOrThrow { repository.findByIdAndWorkspaceId(id, workspaceId) }
            return decryptToModel(entity)
        }

        @Transactional(readOnly = true)
        @PreAuthorize("@workspaceSecurity.hasWorkspace(#workspaceId)")
        fun listByWorkspace(workspaceId: UUID): List<CustomSourceConnectionModel> =
            repository.findByWorkspaceId(workspaceId).map { decryptToModel(it) }

        // ------ Private helpers ------

        private fun runGateChain(host: String, port: Int, database: String, user: String, password: String, sslMode: String) {
            val resolved = ssrfValidator.validateAndResolve(host)
            val primary = resolved.first()
            roVerifier.verify(host, primary, port, database, user, password, sslMode)
        }

        private fun encryptPayload(payload: CredentialPayload): EncryptedCredentials =
            encryptionService.encrypt(objectMapper.writeValueAsString(payload))

        private fun decryptToModel(entity: CustomSourceConnectionEntity): CustomSourceConnectionModel {
            return try {
                val json = encryptionService.decrypt(EncryptedCredentials(entity.encryptedCredentials, entity.iv, entity.keyVersion))
                val payload = objectMapper.readValue(json, CredentialPayload::class.java)
                entity.toModel(payload.host, payload.port, payload.database, payload.user, payload.sslMode.value)
            } catch (e: CryptoException) {
                transitionToFailed(entity, "Configuration error — contact support.")
                buildFailedModel(entity, reason = "Configuration error — contact support.")
            } catch (e: DataCorruptionException) {
                transitionToFailed(entity, "Stored credentials are unreadable — please re-enter the password.")
                buildFailedModel(entity, reason = "Stored credentials are unreadable — please re-enter the password.")
            }
        }

        private fun transitionToFailed(entity: CustomSourceConnectionEntity, reason: String) {
            // Persist status transition; log without credential material
            if (entity.connectionStatus.canTransitionTo(ConnectionStatus.FAILED)) {
                entity.connectionStatus = ConnectionStatus.FAILED
                entity.lastFailureReason = reason
                repository.save(entity)
            }
            logger.warn { "Connection ${entity.id} transitioned to FAILED: $reason" }
        }

        private fun buildFailedModel(entity: CustomSourceConnectionEntity, reason: String): CustomSourceConnectionModel =
            CustomSourceConnectionModel(
                id = requireNotNull(entity.id) { "id required" },
                workspaceId = entity.workspaceId,
                name = entity.name,
                host = "[unavailable]", port = 0, database = "[unavailable]", user = "[unavailable]", sslMode = "[unavailable]",
                connectionStatus = ConnectionStatus.FAILED,
                lastVerifiedAt = entity.lastVerifiedAt,
                createdAt = entity.createdAt, updatedAt = entity.updatedAt,
            )

        private fun toPayload(r: CreateCustomSourceConnectionRequest) =
            CredentialPayload(r.host, r.port, r.database, r.user, r.password, r.sslMode)

        private fun sanitizeForUser(msg: String?): String = msg ?: "verification failed"
    }

    data class TestResult(val pass: Boolean, val category: String?, val message: String)
    ```

    @RestController
    @RequestMapping("/api/v1/custom-sources/connections")
    @Tag(name = "custom-sources")
    class CustomSourceConnectionController(

    ) {
        @PostMapping("/test")
