    ```kotlin
    package riven.core.service.customsource

    interface NameResolver {
        fun resolve(host: String): Array<InetAddress>
    }

    @Component
    class DefaultNameResolver : NameResolver {
        override fun resolve(host: String): Array<InetAddress> = InetAddress.getAllByName(host)
    }

    @Service
    class SsrfValidatorService(
        private val logger: KLogger,
        private val resolver: NameResolver,
    ) {
        fun validateAndResolve(host: String): List<InetAddress> {
            val addresses = try { resolver.resolve(host).toList() }
                catch (e: UnknownHostException) {
                    throw SsrfRejectedException(GENERIC_MESSAGE)
                }
            require(addresses.isNotEmpty()) { GENERIC_MESSAGE }
            for (addr in addresses) {
                checkAddress(addr, host)
            }
            return addresses
        }

        private fun checkAddress(addr: InetAddress, host: String) {
            val blocked = when {
                addr.isLoopbackAddress -> "loopback"
                addr.isLinkLocalAddress -> "link-local"
                addr.isSiteLocalAddress -> "site-local"
                addr.isMulticastAddress -> "multicast"
                addr.isAnyLocalAddress -> "any-local"  // 0.0.0.0
                addr is Inet4Address -> ipv4ExplicitBlock(addr)
                addr is Inet6Address -> ipv6ExplicitBlock(addr)
                else -> null
            }
            if (blocked != null) {
                logger.warn { "SSRF reject: host=$host ip=${addr.hostAddress} category=$blocked" }
                throw SsrfRejectedException(GENERIC_MESSAGE)
            }
        }

        private fun ipv4ExplicitBlock(addr: Inet4Address): String? {
            val ip = BigInteger(1, addr.address).toLong()
            // 100.64.0.0/10 CGNAT
            if (inCidr(ip, 0x64400000L, 10)) return "cgnat"
            // 224.0.0.0/4 multicast (isMulticast already covers, but belt-and-suspenders)
            if (inCidr(ip, 0xE0000000L, 4)) return "multicast"
            // 255.255.255.255 broadcast
            if (ip == 0xFFFFFFFFL) return "broadcast"
            // 0.0.0.0/8 handled by isAnyLocalAddress for 0.0.0.0 exact; cover 0.x.x.x range
            if (inCidr(ip, 0x00000000L, 8)) return "reserved-zero"
            return null
        }

        private fun ipv6ExplicitBlock(addr: Inet6Address): String? {
            val bytes = addr.address
            // IPv4-mapped ::ffff:0:0/96 — unwrap and re-check as IPv4
            if (bytes.slice(0..9).all { it == 0.toByte() } && bytes[10] == 0xFF.toByte() && bytes[11] == 0xFF.toByte()) {
                val v4 = Inet4Address.getByAddress(bytes.sliceArray(12..15)) as Inet4Address
                return if (v4.isLoopbackAddress || v4.isSiteLocalAddress || v4.isLinkLocalAddress ||
                           v4.isAnyLocalAddress || ipv4ExplicitBlock(v4) != null) "mapped-private" else null
            }
            // fc00::/7 ULA
            if ((bytes[0].toInt() and 0xFE) == 0xFC) return "ula"
            return null
        }

        private fun inCidr(ip: Long, network: Long, prefixBits: Int): Boolean {
            val mask = if (prefixBits == 0) 0L else (-1L shl (32 - prefixBits)) and 0xFFFFFFFFL
            return (ip and mask) == (network and mask)
        }

        companion object {
            const val GENERIC_MESSAGE =
                "Host not reachable: this address is blocked for security reasons (private/loopback/metadata)."
        }
    }
    ```

    ```kotlin
    @Service
    class ReadOnlyRoleVerifierService(private val logger: KLogger) {

        fun verify(
            host: String,
            resolvedIp: InetAddress,
            port: Int,
            database: String,
            user: String,
            password: String,
            sslMode: String,
        ) {
            // Connect by resolved IP to prevent DNS rebinding; use sslServerHostname for TLS SNI
            val jdbcUrl = buildJdbcUrl(resolvedIp, port, database, sslMode, host)
            val props = Properties().apply {
                setProperty("user", user)
                setProperty("password", password)
                // sslServerHostname handled in URL for postgresql-jdbc 42.2+
            }

            try {
                DriverManager.getConnection(jdbcUrl, props).use { conn ->
                    conn.autoCommit = false  // required for SAVEPOINT
                    checkSuperuserAttributes(conn)
                    checkWritePrivileges(conn)
                    checkSavepointProbe(conn)
                    conn.rollback()  // no side effects
                }
            } catch (e: ReadOnlyVerificationException) {
                throw e
            } catch (e: SQLException) {
                val sanitized = sanitize(e.message)
                logger.warn { "RO verification JDBC failure: $sanitized" }
                throw ReadOnlyVerificationException("Database connection failed: $sanitized", e)
            }
        }

        private fun buildJdbcUrl(ip: InetAddress, port: Int, database: String, sslMode: String, hostForSni: String): String {
            val ipHost = if (ip is Inet6Address) "[${ip.hostAddress}]" else ip.hostAddress
            // sslhostnameverifier=<fully-qualified-class> not used; sslmode=$sslMode + TLS cert verification happens against sslServerHostname
            return "jdbc:postgresql://$ipHost:$port/$database?sslmode=$sslMode&sslServerHostname=$hostForSni"
        }

        private fun checkSuperuserAttributes(conn: Connection) {
            conn.prepareStatement(
                "SELECT rolsuper, rolcreatedb, rolcreaterole FROM pg_roles WHERE rolname = current_user"
            ).executeQuery().use { rs ->
                if (rs.next() && (rs.getBoolean(1) || rs.getBoolean(2) || rs.getBoolean(3))) {
                    throw ReadOnlyVerificationException(
                        "Role has superuser/createdb/createrole attributes — read-only role required."
                    )
                }
            }
        }

        private fun checkWritePrivileges(conn: Connection) {
            // Count tables in user-accessible schemas where current role has INSERT/UPDATE/DELETE
            // Exclude information_schema + pg_catalog (research pitfall 5)
            val sql = """
                SELECT COUNT(*)
                FROM pg_class c
                JOIN pg_namespace n ON n.oid = c.relnamespace
                WHERE c.relkind IN ('r', 'p')
                  AND n.nspname NOT IN ('information_schema', 'pg_catalog')
                  AND has_schema_privilege(current_user, n.nspname, 'USAGE')
                  AND (has_table_privilege(current_user, c.oid, 'INSERT')
                    OR has_table_privilege(current_user, c.oid, 'UPDATE')
                    OR has_table_privilege(current_user, c.oid, 'DELETE'))
            """.trimIndent()
            conn.prepareStatement(sql).executeQuery().use { rs ->
                if (rs.next()) {
                    val count = rs.getInt(1)
                    if (count > 0) {
                        throw ReadOnlyVerificationException(
                            "Role has write privileges on $count table(s) — read-only role required."
                        )
                    }
                }
            }
        }

        private fun checkSavepointProbe(conn: Connection) {
            val savepoint = conn.setSavepoint("ro_check")
            try {
                // Attempt CREATE TABLE in public — will fail with 42501 for RO role lacking CREATE on public
                conn.prepareStatement("CREATE TABLE __riven_ro_probe_${UUID.randomUUID().toString().replace("-","")} (v int)").execute()
                // If we reach here, write succeeded → reject
                throw ReadOnlyVerificationException("Role can CREATE tables — read-only role required.")
            } catch (e: SQLException) {
                if (e.sqlState == "42501") {
                    // Expected — insufficient_privilege
                    return
                }
                throw e
            } finally {
                try { conn.rollback(savepoint) } catch (_: SQLException) { /* best-effort */ }
            }
        }

        private fun sanitize(message: String?): String =
            message?.replace(Regex("(jdbc:)?postgresql://[^\\s]+"), "[REDACTED_JDBC_URL]") ?: "unknown"
    }
    ```

    ```kotlin
    @Testcontainers
    class ReadOnlyRoleVerifierServiceTest {
        companion object {
            @JvmStatic @Container
            val postgres = PostgreSQLContainer(
                DockerImageName.parse("pgvector/pgvector:pg16").asCompatibleSubstituteFor("postgres")
            ).withDatabaseName("riven_test").withUsername("test").withPassword("test")
        }

        private val service = ReadOnlyRoleVerifierService(mockKLogger())

        @BeforeEach
        fun setupRoles() {
            postgres.createConnection("").use { conn ->
                conn.createStatement().execute("""
                    DROP ROLE IF EXISTS ro_user;
                    CREATE ROLE ro_user LOGIN PASSWORD 'ro_pw';
                    GRANT CONNECT ON DATABASE riven_test TO ro_user;
                    GRANT USAGE ON SCHEMA public TO ro_user;
                    REVOKE CREATE ON SCHEMA public FROM ro_user;
                    GRANT SELECT ON ALL TABLES IN SCHEMA public TO ro_user;

                    DROP ROLE IF EXISTS rw_user;
                    CREATE ROLE rw_user LOGIN PASSWORD 'rw_pw';
                    GRANT CONNECT ON DATABASE riven_test TO rw_user;
                    GRANT USAGE, CREATE ON SCHEMA public TO rw_user;
                    -- create a sample table rw_user has INSERT on
                    CREATE TABLE IF NOT EXISTS public.sample (id int);
                    GRANT INSERT, SELECT ON public.sample TO rw_user;
                """.trimIndent())
            }
        }

        @Test fun `rejects superuser role`() {
            assertThrows<ReadOnlyVerificationException> {
                service.verify(
                    host = "localhost",
                    resolvedIp = InetAddress.getByName(postgres.host),
                    port = postgres.firstMappedPort,
                    database = postgres.databaseName,
                    user = postgres.username,   // 'test' = superuser in Testcontainers
                    password = postgres.password,
                    sslMode = "disable",  // Testcontainers default has no TLS
                )
            }.also { assertThat(it.message).contains("superuser") }
        }

        @Test fun `rejects role with write privileges`() {
            assertThrows<ReadOnlyVerificationException> {
                service.verify("localhost", InetAddress.getByName(postgres.host), postgres.firstMappedPort,
                    postgres.databaseName, "rw_user", "rw_pw", "disable")
            }.also { assertThat(it.message).contains("write privileges") }
        }

        @Test fun `accepts read-only role`() {
            // Should not throw
            service.verify("localhost", InetAddress.getByName(postgres.host), postgres.firstMappedPort,
                postgres.databaseName, "ro_user", "ro_pw", "disable")
        }

        @Test fun `SAVEPOINT probe rolls back with no residue`() {
            // After verify succeeds against ro_user, connect as superuser and confirm no __riven_ro_probe_* table exists
            service.verify("localhost", InetAddress.getByName(postgres.host), postgres.firstMappedPort,
                postgres.databaseName, "ro_user", "ro_pw", "disable")
            postgres.createConnection("").use { conn ->
                val rs = conn.createStatement().executeQuery(
                    "SELECT COUNT(*) FROM pg_tables WHERE tablename LIKE '__riven_ro_probe_%'"
                )
                rs.next(); assertThat(rs.getInt(1)).isEqualTo(0)
            }
        }
    }
    ```
