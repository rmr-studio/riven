enum class MatchSignalType {
    EMAIL, PHONE, NAME, COMPANY, CUSTOM_IDENTIFIER;

    companion object {
        val DEFAULT_WEIGHTS: Map<MatchSignalType, Double> = mapOf(
            EMAIL to 0.9, PHONE to 0.85, NAME to 0.5,

        )

        fun fromSchemaType(schemaType: SchemaType): MatchSignalType = when (schemaType) {
            SchemaType.EMAIL -> EMAIL
            SchemaType.PHONE -> PHONE
            else -> CUSTOM_IDENTIFIER
        }
    }
}

  <action>
    Create `IdentityNormalizationService` as a `@Service` in `riven.core.service.identity` package.

    **Service structure (per CONTEXT.md locked decisions):**
    - Constructor-injected `KLogger` (prototype-scoped bean from LoggerConfig)
    - Single public method: `fun normalize(value: String, signalType: MatchSignalType): String`
    - Internal `when(signalType)` dispatch to private methods: `normalizePhone()`, `normalizeEmail()`, `normalizeText()`
    - Companion object with two hardcoded `Set<String>` constants:
      - `NAME_STOPWORDS`: mr, mrs, ms, dr, prof, rev, sr, jr, ii, iii, iv, phd, md, dds, cpa, esq (~20 terms)
      - `COMPANY_STOPWORDS`: inc, llc, corp, corporation, ltd, limited, gmbh, co, company, lp, llp, plc

    **normalizePhone(value):**
    - Replace all non-digit chars: `value.replace(Regex("[^0-9]"), "")`
    - If result is 11+ digits AND starts with "1", drop leading digit
    - Return digits string

    **normalizeEmail(value):**
    - trim().lowercase()
    - Find '@' index; if no '@', return lowercased value
    - Split local part at '+', take first segment
    - Rejoin local + domain

    **normalizeText(value, stopwords, stripFromStart):**
    - Step 1: `java.text.Normalizer.normalize(value.trim(), Normalizer.Form.NFKD)` then strip combining marks with `Regex("\\p{Mn}")`, then lowercase
    - Step 2: Tokenize on `\s+`, filter empty
    - Step 3: If `stripFromStart`, strip tokens from front while `token.trimEnd('.') in stopwords`
    - Step 4: Strip tokens from end while `token.trimEnd('.') in stopwords`
    - Step 5: `joinToString(" ")`

    **Dispatch:**
    - PHONE -> normalizePhone(value)
    - EMAIL -> normalizeEmail(value)
    - NAME -> normalizeText(value, NAME_STOPWORDS, stripFromStart = true)
    - COMPANY -> normalizeText(value, COMPANY_STOPWORDS, stripFromStart = false)
    - CUSTOM_IDENTIFIER -> value.trim().lowercase()
