<objective>
Implement confidence gate and cross-type scoring discount in IdentityMatchScoringService with comprehensive unit tests.

Purpose: Without the confidence gate, weak single-signal candidates (e.g. a low-similarity NAME-only match) generate noisy match suggestions. Without cross-type discounting, a value matched against the wrong signal type (e.g. an email matched against a name attribute) scores at full weight. These two scoring improvements make match suggestions trustworthy.

Output: Updated IdentityMatchScoringService with passesConfidenceGate() and cross-type discount in buildSignals(), plus unit tests covering all gate and discount behaviors.
</objective>

```kotlin
companion object {
    const val MINIMUM_SCORE_THRESHOLD = 0.5
}

private fun scoreCandidate(...): ScoredCandidate? {
    val signals = buildSignals(triggerAttributes, candidateRows)
    if (signals.isEmpty()) return null
    val compositeScore = computeCompositeScore(signals)
    if (compositeScore < MINIMUM_SCORE_THRESHOLD) { ... return null }
    return ScoredCandidate(...)
}

private fun buildSignals(triggerAttributes, candidateRows): List<MatchSignal> {
    return candidateRows.groupBy { it.signalType }.mapNotNull { (signalType, rowsForType) ->
        val best = rowsForType.maxByOrNull { it.similarityScore } ?: return@mapNotNull null
        if (best.similarityScore == 0.0) return@mapNotNull null
        val sourceValue = triggerAttributes[signalType] ?: best.candidateValue
        val weight = MatchSignalType.DEFAULT_WEIGHTS[signalType] ?: error(...)
        MatchSignal(type = signalType, sourceValue = sourceValue, targetValue = best.candidateValue, similarity = best.similarityScore, weight = weight)
    }
}
```
